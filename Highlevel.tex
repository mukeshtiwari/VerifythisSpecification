   ( In f (dom keys) -> Exists (v : Key), Some v /\ v = keys f) /\ 
   (~In f (dom keys) -> None) 
 


byKeyId (keyId : KeyId) :=
  Exists (l : List[Key]), (forall (x : Key), In x l -> 
     Exists (f : Fingerprint), keys f = x) /\
     (forall (f : Fingerprint), (keys f).keyId = keyId -> In (keys f) l)

        uploaded' = uploaded [token := key.fingerprint]) /\
